data <- mtcars
str(data)
summary(data)
print(colSums(is.na(data)))
for (col in colnames(data)) {
  if (is.numeric(data[[col]])) {
    data[[col]][is.na(data[[col]])] <- mean(data[[col]], na.rm = TRUE)
  }
}
data$cyl <- as.factor(data$cyl)
data$gear <- as.factor(data$gear)
data <- subset(data, select = -c(carb))
numeric_cols <- sapply(data, is.numeric)
for (col in names(data)[numeric_cols]) {
  Q1 <- quantile(data[[col]], 0.25)
  Q3 <- quantile(data[[col]], 0.75)
  IQR <- Q3 - Q1
  lower <- Q1 - 1.5 * IQR
  upper <- Q3 + 1.5 * IQR
  data <- data[data[[col]] >= lower & data[[col]] <= upper, ]
}
for (col in names(data)[numeric_cols]) {
  data[[paste0(col, "_norm")]] <- 
    (data[[col]] - min(data[[col]])) / (max(data[[col]]) - min(data[[col]]))
}
for (col in names(data)[numeric_cols]) {
  data[[paste0(col, "_scaled")]] <- scale(data[[col]])
}
set.seed(123)
index <- sample(1:nrow(data), 0.7 * nrow(data))
train <- data[index, ]
test  <- data[-index, ]
print(head(train))
print(head(test))

library(e1071)     # For SVM
library(caret)     # For confusion matrix
library(ggplot2)   # For visualization
data(iris)
df <- iris
print(head(df))
set.seed(123)
train_index <- sample(1:nrow(df), 0.8 * nrow(df))
train <- df[train_index, ]
test  <- df[-train_index, ]
svm_model <- svm(Species ~ ., data = train,
                 kernel = "radial",   # RBF kernel (most commonly used)
                 cost = 1,
                 gamma = 0.1)
print(svm_model)
pred <- predict(svm_model, newdata = test)
conf <- confusionMatrix(pred, test$Species)
print(conf)
ggplot(df, aes(Petal.Length, Petal.Width, color = Species)) +
  geom_point(size = 3) +
  ggtitle("Iris Dataset - SVM Classification (Petal Length vs Width)") +
  theme_minimal()

  c("Milk", "Bread", "Butter"),
  c("Bread", "Butter"),

  c("Milk", "Butter"),

library(tm)
library(SnowballC)
library(wordcloud)
library(RColorBrewer)
text_data <- c(
"Text mining is an interesting area of data science",
"R is widely used for text mining and data analysis",
"Text mining helps extract useful information from text",
"Data science includes machine learning and text mining"
)
corpus <- Corpus(VectorSource(text_data))
corpus <- tm_map(corpus, content_transformer(tolower))
corpus <- tm_map(corpus, removePunctuation)
corpus <- tm_map(corpus, removeNumbers)
corpus <- tm_map(corpus, removeWords, stopwords("english"))
corpus <- tm_map(corpus, stemDocument)
corpus <- tm_map(corpus, stripWhitespace)
dtm <- DocumentTermMatrix(corpus)
dtm_matrix <- as.matrix(dtm)
word_freq <- sort(colSums(dtm_matrix), decreasing = TRUE)
word_freq_df <- data.frame(
  word = names(word_freq),
  frequency = word_freq
)
print(word_freq_df)
set.seed(123)
wordcloud(
  words = word_freq_df$word,
  freq = word_freq_df$frequency,
  min.freq = 1,
  max.words = 50,
  random.order = FALSE,
  colors = brewer.pal(8, "Dark2")
)
